
GFEXT_planet_pop_assembly = {
	add = modifier:planet_pop_assembly_add
	mult = value:scripted_modifier_mult|MODIFIER|planet_pop_assembly_mult|
}

# 计算方式：
#       doll pops
# ---------------------- * modifier:planet_pop_assembly_add * cost
#   total robotic pops
# this = planet
GFEXT_doll_pops_assembling_cost = {
	complex_trigger_modifier = {
		trigger = count_owned_pop_amount
		parameters = { limit = { has_trait = GF_trait_doll } }
		mode = add
	}
	complex_trigger_modifier = {
		trigger = count_owned_pop_amount
		parameters = {
			limit = { is_robot_pop_group = yes }
		}
		mode = divide
	}
	mult = modifier:planet_pop_assembly_add
	mult = @GF_doll_pops_assembling_gf_core_chip_cost
	min = 0
}

GFEXT_planet_assembly_job_count = {
	complex_trigger_modifier = {
		trigger = num_pops_with_job_tag
		parameters = {
			tags = { pop_assembly }
		}
		mode = add
	}
	min = 1
	[[FACTOR]
		mult = $FACTOR$
	]
}

GFEXT_doll_pops_assembly_job_upkeep = {
	add = value:GFEXT_doll_pops_assembling_cost
	mult = 100

	# 然而，这里有非常大的问题：
	# 如果一个人口获得了基础的生产力加成，例如一个人口实际提供2点劳动力，那么50人即可提供100人的劳动力
	# 然后这里除以0.5，于是计算得到的值就比正确值大了一倍
	# 理论上来说，这里应该除以的是组装岗上的劳动力总和
	complex_trigger_modifier = {
		trigger = num_pops_with_job_tag
		parameters = {
			tags = { pop_assembly }
		}
		mode = divide
	}
}
GFEXT_doll_pops_assembly_district_upkeep = {
	add = value:GFEXT_doll_pops_assembling_cost
	complex_trigger_modifier = {
		trigger = num_districts
		parameters = {
			type = $district$
		}
		mode = divide
	}
}
